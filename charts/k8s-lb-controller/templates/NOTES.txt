1. Confirm the controller is running:
   kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/name={{ include "k8s-lb-controller.name" . }}

{{- if .Values.dataplane.enabled }}

2. Confirm the dataplane pod and service are running:
   kubectl get deploy,svc -n {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=dataplane

   Controller dataplane URL:
   {{ include "k8s-lb-controller.controllerDataplaneAPIURL" . }}

   Dataplane host interface:
   {{ .Values.dataplane.interface }}

   Dataplane IP attach mode:
   {{ .Values.dataplane.ipAttach.mode }}

{{- else }}

2. Controller provider mode:
   {{ .Values.controller.providerMode }}

{{- end }}

3. Example managed Service:

   apiVersion: v1
   kind: Service
   metadata:
     name: demo
   spec:
     type: LoadBalancer
     loadBalancerClass: {{ .Values.controller.loadBalancerClass }}
     selector:
       app: demo
     ports:
       - port: 80
         targetPort: 80

4. Watch the assigned external address:
   kubectl get svc demo -o wide
